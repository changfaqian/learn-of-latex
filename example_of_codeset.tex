inline int gcd(int a, int b) { 
// 如果a<b，则递归得gcd(b,a%b)即gcd(b, a)，即交换了位置，时间复杂度O(log max(a, b))
return b==0?a:gcd(b,a%b)
}
inline int lcm(int a, int b) {
return a/gcd(a,b)*b;
}
